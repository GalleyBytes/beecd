1. Get the Hive HQ web interface:
{{- if .Values.hiveHq.ingress.enabled }}
{{- range $host := .Values.hiveHq.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.hiveHq.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "ClusterIP" .Values.hiveHq.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "beecd.hiveHq.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:8080
  echo "Visit http://127.0.0.1:8080"
{{- else if contains "LoadBalancer" .Values.hiveHq.service.type }}
  kubectl get svc --namespace {{ .Release.Namespace }} {{ include "beecd.fullname" . }}-hive-hq
{{- end }}

2. Add a cluster (agent deployment) using Hive HQ:
  - Open Hive HQ
  - Use "+ Add Cluster" to generate the agent manifest
  - Apply the generated manifest to the target cluster

3. Label namespaces for BeeCD to manage:
  kubectl label namespace <your-namespace> beecd/register=true

4. Check status:
  kubectl --namespace {{ .Release.Namespace }} get pods
